114 uint8_t TwoWire::requestFrom(uint8_t address, uint8_t quantity, uint32_t iaddress, uint8_t isize, uint8_t sendStop)
115 {
116  if (isize > 0) {
117  // send internal address; this mode allows sending a repeated start to access
118  // some devices' internal registers. This function is executed by the hardware
119  // TWI module on other processors (for example Due's TWI_IADR and TWI_MMR registers)
120 
121  beginTransmission(address);
122 
123  // the maximum size of internal address is 3 bytes
124  if (isize > 3){
125  isize = 3;
126  }
127 
128  // write internal register address - most significant byte first
129  while (isize-- > 0)
130  write((uint8_t)(iaddress >> (isize*8)));
131  endTransmission(false);
132  }
133 
134  // clamp to buffer length
135  if(quantity > BUFFER_LENGTH){
136  quantity = BUFFER_LENGTH;
137  }
138  // perform blocking read into buffer
139  uint8_t read = twi_readFrom(address, rxBuffer, quantity, sendStop);
140  // set rx buffer iterator vars
141  rxBufferIndex = 0;
142  rxBufferLength = read;
143 
144  return read;
145 }

182 //
183 // Originally, 'endTransmission' was an f(void) function.
184 // It has been modified to take one parameter indicating
185 // whether or not a STOP should be performed on the bus.
186 // Calling endTransmission(false) allows a sketch to
187 // perform a repeated start.
188 //
189 // WARNING: Nothing in the library keeps track of whether
190 // the bus tenure has been properly ended with a STOP. It
191 // is very possible to leave the bus in a hung state if
192 // no call to endTransmission(true) is made. Some I2C
193 // devices will behave oddly if they do not see a STOP.
194 //
195 uint8_t TwoWire::endTransmission(uint8_t sendStop)
196 {
197  // transmit buffer (blocking)
198  uint8_t ret = twi_writeTo(txAddress, txBuffer, txBufferLength, 1, sendStop);
199  // reset tx buffer iterator vars
200  txBufferIndex = 0;
201  txBufferLength = 0;
202  // indicate that we are done transmitting
203  transmitting = 0;
204  return ret;
205 }

218 size_t TwoWire::write(uint8_t data)
219 {
220  if(transmitting){
221  // in master transmitter mode
222  // don't bother if buffer is full
223  if(txBufferLength >= BUFFER_LENGTH){
224  setWriteError();
225  return 0;
226  }
227  // put byte in tx buffer
228  txBuffer[txBufferIndex] = data;
229  ++txBufferIndex;
230  // update amount in buffer
231  txBufferLength = txBufferIndex;
232  }else{
233  // in slave send mode
234  // reply to master
235  twi_transmit(&data, 1);
236  }
237  return 1;
238 }

301 // behind the scenes function that is called when data is received
302 void TwoWire::onReceiveService(uint8_t* inBytes, int numBytes)
303 {
304  // don't bother if user hasn't registered a callback
305  if(!user_onReceive){
306  return;
307  }
308  // don't bother if rx buffer is in use by a master requestFrom() op
309  // i know this drops data, but it allows for slight stupidity
310  // meaning, they may not have read all the master requestFrom() data yet
311  if(rxBufferIndex < rxBufferLength){
312  return;
313  }
314  // copy twi rx buffer into local read buffer
315  // this enables new reads to happen in parallel
316  for(uint8_t i = 0; i < numBytes; ++i){
317  rxBuffer[i] = inBytes[i];
318  }
319  // set rx iterator vars
320  rxBufferIndex = 0;
321  rxBufferLength = numBytes;
322  // alert user program
323  user_onReceive(numBytes);
324 }
