888  int16_t data;

891

893  write(address);
894  crc_update(address);
895  write(GETPINFUNCTIONS);

913 
914  if(data!=-1){

916  data = read(timeout);
917  if(data!=-1){
918  ccrc = data << 8;
919  data = read(timeout);

923  S3mode = val1;
924  S4mode = val2;
925  S5mode = val3;
